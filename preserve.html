<script>

function buildCandles() {

  const bg = document.getElementById('candlesBg');

  for (let i = 0; i < 20; i++) {

    const wrap = document.createElement('div');

    wrap.className = 'c-wrap';

    const delay = Math.random() * 1.5;

    const dur = 1 + Math.random();

    const h = 14 + Math.random() * 16;

    wrap.innerHTML = `

      <div class="c-flame" style="animation-delay:${delay}s;animation-duration:${dur}s"></div>

      <div class="c-body" style="height:${h}px"></div>

    `;

    bg.appendChild(wrap);

  }

}


function openModal(tier, krw, usd, period) {

  document.getElementById('modalTier').textContent = `Preserve for ${period}`;

  document.getElementById('modalPrice').textContent = usd;

  document.getElementById('modalPriceSub').textContent = 'one-time';

  document.getElementById('modalOverlay').classList.add('show');

}


function closeModal() {

  document.getElementById('modalOverlay').classList.remove('show');

  document.getElementById('emailInput').value = '';

}


function closeModalOutside(e) {

  if (e.target === document.getElementById('modalOverlay')) closeModal();

}


function submitEmail() {

  const email = document.getElementById('emailInput').value.trim();

  if (!email || !email.includes('@')) {

    showToast('Please enter a valid email');

    return;

  }

  closeModal();

  showToast('✓ We will notify you when preservation opens 🏮');

}


function showToast(msg) {

  const t = document.getElementById('toast');

  t.textContent = msg;

  t.classList.add('show');

  setTimeout(() => t.classList.remove('show'), 3000);

}


buildCandles();

</script>
